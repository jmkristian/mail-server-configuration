Here's how the packages were installed, as recommended by:
https://www.vultr.com/docs/how-to-install-postfix-dovecot-and-roundcube-on-ubuntu-20-04

apt-get install -y postfix
# select "Internet Site"
# mail domain name: jkristian.com

apt-get install -y dovecot-core dovecot-imapd dovecot-pop3d dovecot-lmtpd
# By the way, here's how to un-install it completely (including configuration files):
# apt-get purge -y dovecot-core dovecot-imapd dovecot-pop3d dovecot-lmtpd
# apt-get autoremove --purge -y

groupadd -g 5000 vmail
useradd -r -g vmail -u 5000 vmail -d /var/mail/vhosts -c "virtual mail user"
mkdir -p /var/mail/vhosts/jkristian.com
chown -R vmail:vmail /var/mail/vhosts/
# Create /etc/dovecot/users, containing:
getmail@jkristian.com:{plain}STRONG_PASSWORD
# ... with a long unguessable password in place of STRONG_PASSWORD.

# Deploy the new configuration files (./deploy).
# Test:
service dovecot start; service postfix start
telnet `hostname` 587
ehlo jkristian.com
quit
# Configure an MUA (e.g. Outlook or Thunderbird) to authenticate
# to `hostname` as getmail@jkristian.com with STRONG_PASSWORD.
service postfix stop; service dovecot stop

####################################################################
# An alternative is Cyrus/SASL authentication. It's more complicated.
# Uncomment the relevant lines in /etc/postfix/main.cf,
# and comment out the lines for dovecot.
apt-get install -y sasl2-bin
adduser postfix sasl
# Edit /etc/default/saslauthd to contain:
START=yes
MECHANISMS="sasldb"
OPTIONS="-c -m /var/run/saslauthd"

saslpasswd2 -c -u `postconf -h mydomain` getmail@jkristian.com
# Enter a long unguessable password when prompted.
# Test:
service saslauthd start
testsaslauthd -u getmail -r jkristian.com -p STRONG_PASSWORD
# ... with the password you entered to saslpasswd2 in place of STRONG_PASSWORD.
